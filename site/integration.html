Cross-Border Data Transfer Compliance

Compare privacy requirements between countries before transferring data internationally.

async function validateDataTransfer(fromCountry, toCountry, dataFields) {
  const comparison = await openpii.compareCountries([fromCountry, toCountry]);
  
  const violations = [];
  for (const field of dataFields) {
    const sourceReqs = comparison[fromCountry].getRequirements(field);
    const destReqs = comparison[toCountry].getRequirements(field);
    
    if (destReqs.isMoreRestrictive(sourceReqs)) {
      violations.push({
        field,
        issue: 'Destination has stricter requirements',
        recommendation: destReqs.getRecommendation()
      });
    }
  }
  
  return { isCompliant: violations.length === 0, violations };
}

class OpenPIIMapCache {
    constructor(client, options = {}) {
        this.client = client;
        this.cache = new Map();
        this.ttl = options.ttl || 3600000; // 1 hour
        this.maxSize = options.maxSize || 1000;
    }
    
    async getCountry(countryId) {
        const cacheKey = `country:${countryId}`;
        const cached = this.cache.get(cacheKey);
        
        if (cached && Date.now() - cached.timestamp < this.ttl) {
            return cached.data;
        }
        
        const data = await this.client.getCountry(countryId);
        this.cache.set(cacheKey, {
            data,
            timestamp: Date.now()
        });
        
        // Cleanup old entries
        if (this.cache.size > this.maxSize) {
            this.cleanup();
        }
        
        return data;
    }
    
    cleanup() {
        const now = Date.now();
        for (const [key, value] of this.cache.entries()) {
            if (now - value.timestamp > this.ttl) {
                this.cache.delete(key);
            }
        }
    }
}

Error Handling & Resilience

Implement robust error handling and fallback mechanisms.

func (c *Client) GetCountryWithRetry(countryID string) (*Country, error) {
    var lastErr error
    
    for attempt := 0; attempt < c.maxRetries; attempt++ {
        country, err := c.GetCountry(countryID)
        if err == nil {
            return country, nil
        }
        
        lastErr = err
        
        // Check if error is retryable
        if !isRetryableError(err) {
            return nil, err
        }
        
        // Exponential backoff
        waitTime := time.Duration(math.Pow(2, float64(attempt))) * time.Second
        time.Sleep(waitTime)
        
        log.Printf("Retry attempt %d/%d for country %s", attempt+1, c.maxRetries, countryID)
    }
    
    return nil, fmt.Errorf("failed after %d attempts: %w", c.maxRetries, lastErr)
}

func isRetryableError(err error) bool {
    // Check for network errors, timeouts, 5xx status codes
    var netErr net.Error
    if errors.As(err, &netErr) && netErr.Timeout() {
        return true
    }
    
    var httpErr *HTTPError
    if errors.As(err, &httpErr) {
        return httpErr.StatusCode >= 500
    }
    
    return false
}

Implementation Patterns
Synchronous Pattern

Real-time compliance checks during data processing.

  • Immediate validation
  • Higher latency
  • Real-time accuracy
  • Network dependent
Asynchronous Pattern

Queue-based processing with eventual compliance.

  • Better performance
  • Eventual consistency
  • Offline capability
  • Batch processing

Best Practices

Security Best Practices
Store API keys in environment variables or secure vaults
Use HTTPS for all API communications
Implement rate limiting on your side to respect API limits
Log API interactions for audit purposes
Performance Best Practices
Cache country data locally with appropriate TTL
Use batch API calls when processing multiple countries
Implement connection pooling for high-volume applications
Consider offline-first architecture for critical applications
Common Pitfalls
  • Not handling API rate limits properly
  • Caching data too long without checking for updates
  • Ignoring network timeouts and error handling
  • Hard-coding country codes instead of using dynamic lookup

import logging
import time

class OpenPIIMapMiddleware:
    def __init__(self, client):
        self.client = client
        self.logger = logging.getLogger('openpiimap')
    
    def get_country(self, country_id):
        start_time = time.time()
        try:
            result = self.client.get_country(country_id)
            duration = time.time() - start_time
            
            self.logger.info(
                'API call successful',
                extra={
                    'endpoint': f'/countries/{country_id}',
                    'duration_ms': duration * 1000,
                    'cache_hit': hasattr(result, '_from_cache')
                }
            )
            return result

                'API call failed',
                extra={
